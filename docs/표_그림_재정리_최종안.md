# analysis/create_key_figures.py
"""
핵심 그림 제작 스크립트
사용법: python create_key_figures.py

import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd
import numpy as np

# 한글 폰트 설정
plt.rcParams['font.family'] = 'AppleGothic'  # macOS
plt.rcParams['axes.unicode_minus'] = False

def create_figure_7_5():
    """[그림 7-5] C2 세부 요소 비교"""
    elements = ['C2-1\n사고과정', 'C2-2\n이해도확인', 
                'C2-3\n메타인지', 'C2-4\n깊은사고']
    agent = [48, 52, 31, 39]
    freepass = [36, 41, 27, 34]
    p_values = [0.027, 0.041, 0.445, 0.385]
    
    x = np.arange(len(elements))
    width = 0.35
    
    fig, ax = plt.subplots(figsize=(10, 6))
    bars1 = ax.bar(x - width/2, agent, width, label='Agent', color='#4A90E2')
    bars2 = ax.bar(x + width/2, freepass, width, label='Freepass', color='#E24A4A')
    
    # p-value 표시
    for i, p in enumerate(p_values):
        if p < 0.05:
            ax.text(i, max(agent[i], freepass[i]) + 3, 
                   f'p={p:.3f}*', ha='center', fontsize=9)
    
    ax.set_xlabel('C2 세부 요소', fontsize=12)
    ax.set_ylabel('충족률 (%)', fontsize=12)
    ax.set_title('[그림 7-5] C2(학습 지원) 세부 요소 비교 (Agent vs Freepass)', 
                fontsize=14, fontweight='bold')
    ax.set_xticks(x)
    ax.set_xticklabels(elements)
    ax.legend()
    ax.grid(axis='y', alpha=0.3)
    
    plt.tight_layout()
    plt.savefig('docs/assets/images/chapter7/fig-7-5-c2-elements.png', 
                dpi=300, bbox_inches='tight')
    print("✅ 그림 7-5 생성 완료")

    # p-value와 효과크기 표시
